(static) build(source, tonic) → {Array}

Create chords by chord type or intervals and tonic. The returned chord is an array of notes (or intervals if you specify false as tonic)

This function is currified

Parameters:
Name Type Description
source String

the chord type, intervals or notes
tonic String

the chord tonic (or false to get intervals)
Returns:

the chord notes

Type
Array
Example
var chord = require('tonal-chord')
// get chord notes using type and tonic
chord.build('maj7', 'C2') // => ['C2', 'E2', 'G2', 'B2']
// get chord intervals (tonic false)
chord.build('maj7', false) // => ['1P', '3M', '5P', '7M']
// partially applied
var maj7 = chord.build('maj7')
maj7('C') // => ['C', 'E', 'G', 'B']
// create chord from intervals
chord.build('1 3 5 m7 m9', 'C') // => ['C', 'E', 'G', 'Bb', 'Db']

(static) get(name) → {Array}

Get chord notes from chord name

Parameters:
Name Type Description
name String

the chord name
Returns:

the chord notes

Type
Array
Example
import { fromName } from 'tonal-chords'
fromName('C7') // => ['C', 'E', 'G', 'Bb']
fromName('CMaj7') // => ['C', 'E', 'G', 'B']
